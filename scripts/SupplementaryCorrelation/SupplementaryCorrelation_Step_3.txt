def benjamini_hochberg(p_values):
    """
    Benjamini-Hochberg FDR korekcija p-vrijednosti.
    Funkcija dopušta NaN vrijednosti i vraća NaN na istim mjestima.
    """
    p_values = np.asarray(p_values, dtype=float)
    adjusted = np.full(len(p_values), np.nan)

    valid_mask = ~np.isnan(p_values)
    valid_p = p_values[valid_mask]

    if len(valid_p) == 0:
        return adjusted

    n = len(valid_p)
    order = np.argsort(valid_p)
    ranked_p = valid_p[order]

    adjusted_valid = np.empty(n, dtype=float)
    cumulative_min = 1.0

    for i in range(n - 1, -1, -1):
        rank = i + 1
        value = ranked_p[i] * n / rank
        cumulative_min = min(cumulative_min, value)
        adjusted_valid[order[i]] = cumulative_min

    adjusted[valid_mask] = np.minimum(adjusted_valid, 1.0)

    return adjusted


def phi_coefficient_from_table(a, b, c, d):
    """
    Izračun phi koeficijenta za 2x2 tablicu:

                     motive=1    motive=0
        funding=1       a            b
        funding=0       c            d
    """
    denominator = np.sqrt(
        (a + b) * (c + d) * (a + c) * (b + d)
    )

    if denominator == 0:
        return np.nan

    return (a * d - b * c) / denominator


def cliffs_delta(x, y):
    """
    Cliffov delta za dvije nezavisne skupine.
    Pozitivna vrijednost označava više vrijednosti u skupini x.
    """
    x = np.asarray(x)
    y = np.asarray(y)

    n_x = len(x)
    n_y = len(y)

    if n_x == 0 or n_y == 0:
        return np.nan

    greater = 0
    less = 0

    for value in x:
        greater += np.sum(value > y)
        less += np.sum(value < y)

    return (greater - less) / (n_x * n_y)


def cliffs_delta_magnitude(delta):
    """
    Interpretacija apsolutne vrijednosti Cliffova delta koeficijenta.
    """
    if pd.isna(delta):
        return np.nan

    absolute_delta = abs(delta)

    if absolute_delta < 0.147:
        return "negligible"
    elif absolute_delta < 0.330:
        return "small"
    elif absolute_delta < 0.474:
        return "medium"
    else:
        return "large"


def round_df(data):
    """
    Zaokruživanje numeričkih rezultata za čitljiv prikaz i Excel izlaz.
    """
    output = data.copy()

    columns_to_round = [
        "spearman_rho",
        "spearman_p",
        "spearman_p_fdr",
        "kendall_tau_b",
        "kendall_p",
        "kendall_p_fdr",
        "pearson_r",
        "pearson_p",
        "mean_motivators_selected",
        "mean_motivators_not_selected",
        "median_motivators_selected",
        "median_motivators_not_selected",
        "mean_difference_selected_minus_not",
        "mannwhitney_p",
        "mannwhitney_p_fdr",
        "cliffs_delta",
        "p_motive_given_funding",
        "p_motive_given_no_funding",
        "risk_difference",
        "phi",
        "odds_ratio",
        "fisher_p",
        "fisher_p_fdr",
        "chi2",
        "chi2_p",
        "min_expected",
    ]

    for col in columns_to_round:
        if col in output.columns:
            output[col] = output[col].round(4)

    return output

# 3. Glavne korelacije između agregiranih indeksa
#
# Spearmanov rho:
# - glavni rangovni pokazatelj povezanosti
#
# Kendallov tau-b:
# - dopunski rangovni pokazatelj
# - prikladan zbog brojnih vezanih rangova u diskretnim indeksima
#
# Pearsonov r:
# - računa se samo informativno
# - ne treba ga isticati u glavnom tekstu rada
# ------------------------------------------------------------

aggregate_pairs = [
    ("External_funding_count", "Improvement_motivators_count"),
    ("Any_external_funding", "Improvement_motivators_count"),
    ("No_external_funding_indicator", "Improvement_motivators_count"),
    ("Any_EU_or_public_funding", "Improvement_motivators_count"),
    ("Any_bank_or_credit_support", "Improvement_motivators_count"),
    ("External_funding_count", "Any_positive_improvement_motivation"),
    ("External_funding_count", "No_improvement_indicator"),
    ("Any_external_funding", "Any_positive_improvement_motivation"),
    ("Any_external_funding", "No_improvement_indicator"),
]

aggregate_rows = []

for x_col, y_col in aggregate_pairs:
    x = df[x_col]
    y = df[y_col]

    if x.nunique(dropna=True) < 2 or y.nunique(dropna=True) < 2:
        spearman_rho = np.nan
        spearman_p = np.nan
        kendall_tau_b = np.nan
        kendall_p = np.nan
        pearson_r = np.nan
        pearson_p = np.nan
    else:
        spearman_rho, spearman_p = stats.spearmanr(
            x,
            y,
            nan_policy="omit"
        )

        # scipy.stats.kendalltau računa Kendallov tau-b,
        # koji korigira vezane rangove.
        kendall_tau_b, kendall_p = stats.kendalltau(
            x,
            y,
            nan_policy="omit"
        )

        pearson_r, pearson_p = stats.pearsonr(x, y)

    aggregate_rows.append({
        "x_variable": x_col,
        "y_variable": y_col,
        "n": len(df),
        "spearman_rho": spearman_rho,
        "spearman_p": spearman_p,
        "kendall_tau_b": kendall_tau_b,
        "kendall_p": kendall_p,
        "pearson_r": pearson_r,
        "pearson_p": pearson_p,
    })

aggregate_correlations = pd.DataFrame(aggregate_rows)

# FDR korekcija za širu obitelj agregiranih Spearmanovih provjera.
aggregate_correlations["spearman_p_fdr"] = benjamini_hochberg(
    aggregate_correlations["spearman_p"].values
)

aggregate_correlations["spearman_significant_fdr_0_05"] = (
    aggregate_correlations["spearman_p_fdr"] < 0.05
)

# FDR korekcija za širu obitelj agregiranih Kendallovih provjera.
aggregate_correlations["kendall_p_fdr"] = benjamini_hochberg(
    aggregate_correlations["kendall_p"].values
)

aggregate_correlations["kendall_significant_fdr_0_05"] = (
    aggregate_correlations["kendall_p_fdr"] < 0.05
)

# Izdvajanje glavne unaprijed definirane dodatne korelacije.
main_correlation = aggregate_correlations[
    (aggregate_correlations["x_variable"] == "External_funding_count") &
    (aggregate_correlations["y_variable"] == "Improvement_motivators_count")

group_rows = []

for predictor in funding_binary_predictors:
    selected = df.loc[
        df[predictor] == 1,

    not_selected = df.loc[
        df[predictor] == 0,

    group_rows.append({

        "mean_motivators_selected": selected.mean(),
        "mean_motivators_not_selected": not_selected.mean(),
        "median_motivators_selected": selected.median(),
        "median_motivators_not_selected": not_selected.median(),
        "mean_difference_selected_minus_not": (
            selected.mean() - not_selected.mean()
        ),
        "mannwhitney_u": mannwhitney_u,
        "mannwhitney_p": mannwhitney_p,
        "cliffs_delta": delta,
        "cliffs_delta_magnitude": cliffs_delta_magnitude(delta),
    })

funding_group_comparisons = pd.DataFrame(group_rows)

)

# ------------------------------------------------------------
# 5. Pairwise q34 × q31 binarne asocijacije
#
# Za svaki pojedinačni izvor financiranja i svaki pojedinačni
# pozitivan motiv računamo:
# - phi koeficijent
# - Fisherov egzaktni test
# - odds ratio
# - risk difference
#
# Fisherov test je glavni test za ovaj dio jer pojedine kategorije
# imaju male frekvencije.
# ------------------------------------------------------------

pairwise_rows = []

for funding_col in q34_pairwise_cols:
    for motive_col in q31_pairwise_cols:
        funding = df[funding_col].astype(int)
        motive = df[motive_col].astype(int)

        a = int(((funding == 1) & (motive == 1)).sum())
        b = int(((funding == 1) & (motive == 0)).sum())
        c = int(((funding == 0) & (motive == 1)).sum())
        d = int(((funding == 0) & (motive == 0)).sum())

        contingency_table = np.array([
            [a, b],
            [c, d]
        ])

        phi = phi_coefficient_from_table(a, b, c, d)

        odds_ratio, fisher_p = stats.fisher_exact(
            contingency_table,
            alternative="two-sided"
        )

        try:
            chi2, chi2_p, dof, expected = stats.chi2_contingency(
                contingency_table,
                correction=False
            )
            min_expected = expected.min()
        except Exception:
            chi2 = np.nan
            chi2_p = np.nan
            min_expected = np.nan

        funding_n = int(funding.sum())
        motive_n = int(motive.sum())

        p_motive_given_funding = (
            a / (a + b) if (a + b) > 0 else np.nan
        )

        p_motive_given_no_funding = (
            c / (c + d) if (c + d) > 0 else np.nan
        )

        pairwise_rows.append({
            "funding_variable": funding_col,
            "funding_label": q34_labels[funding_col],
            "motive_variable": motive_col,
            "motive_label": q31_labels[motive_col],
            "a_both_1": a,
            "b_funding1_motive0": b,
            "c_funding0_motive1": c,
            "d_both_0": d,
            "funding_n": funding_n,
            "motive_n": motive_n,
            "p_motive_given_funding": p_motive_given_funding,
            "p_motive_given_no_funding": p_motive_given_no_funding,
            "risk_difference": (
                p_motive_given_funding -
                p_motive_given_no_funding
            ),
            "phi": phi,
            "odds_ratio": odds_ratio,
            "fisher_p": fisher_p,
            "chi2": chi2,
            "chi2_p": chi2_p,
            "min_expected": min_expected,
        })

pairwise_associations = pd.DataFrame(pairwise_rows)

pairwise_associations["fisher_p_fdr"] = benjamini_hochberg(
    pairwise_associations["fisher_p"].values
)

pairwise_associations["significant_fdr_0_05"] = (
    pairwise_associations["fisher_p_fdr"] < 0.05
)
